\newcommand\importantstuff[3]{
    \providecommand\xoffset{0.3}
    \providecommand\yoffset{-0.2}
    % Blur
    \foreach \x in {-0.1,0.1} {
        \foreach \y in {-0.1,0.1} {         
            \node[black!20!color2,align=center] at (#1em+\xoffset em+\x em,#2em+\yoffset em+\y em) {
                \scalebox{2}{\Huge\textbf{#3}} 
            };
        }
    }

    % Main Shadow
    \node[black!40!color2,align=center] at (#1em+0.3em,#2em-0.2em) {
        \scalebox{2}{\Huge\textbf{#3}} 
    };
    \node[white,align=center] at (#1em,#2em) {
        \scalebox{2}{\Huge\textbf{#3}} 
    };
}
